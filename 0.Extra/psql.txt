create database filelock;

CREATE TABLE users(
    id SERIAL PRIMARY KEY NOT NULL,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(50) NOT NULL,
    password VARCHAR(50) NOT NULL
); 

CREATE TABLE data(
    id SERIAL PRIMARY KEY NOT NULL,
    name VARCHAR(500) NOT NULL,
    path VARCHAR(1000) NOT NULL,
    email VARCHAR(50) NOT NULL,
    bin_data bytea
);
